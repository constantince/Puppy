const setMPA = () => {
    const entry = {};
    const entryFiels = glob.sync(path.join(__dirname, './src/*/index.{js,tsx,ts}'));
    const plugins = entryFiels.map((v) => {
        const result = v.match(/src\/([^/]+)\/index\.[jt]sx?$/);
        const folderName = result[1];
        entry[folderName] = `./${result[0]}`;
        return new HtmlWebpackPlugin({
            template: path.join(__dirname, `src/${folderName}/index.html`),
            filename: `${folderName}.html`,
            chunks: [folderName, 'vendor'],
            inject: true,
            minify: {
                html5: true,
                removeComments: true,
            },
        });
    });
 
    return {
        entry,
        plugins,
    };
};
